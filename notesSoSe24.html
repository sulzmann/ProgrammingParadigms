{-

List types in Haskell.

[Int]  denotes the type which is a list of Integers

What is required to sort a list?

The elements must be comparable.

Haskell has full type inference!!!!!


Let's take a closer look at the following type

Ord a => [a] -> [a]

^^^^^

"Ord a" is a precondition and states
that the elements of type "a" must be comparable.

"a" is a type parameter.

Short summary:

The type

Ord a => [a] -> [a]

expresses the following logical statement.

forall a. if Ord(a) is satisfied
          then quicksort takes as an input a value of type [a]
          and yields a value of type a]


Take note.

"Ord" is a predicate.

It is a unary predicate, taking as arguments types.

"Ord a" means that the type "a" is comparable.

-}

-- quicksort :: Ord a => [a] -> [a]
--
-- quicksort ::  Ord Int => [Int] -> [Int]
--  The "Ord Int" is redundant, cause "Ord Int" hold.
-- To use such a type annotation, need to enable "FlexibleContexts"

-- As we said,
-- Haskell is expression-oriented.
-- When writing Haskell programs, we are defining a sequence of equations.

quicksort [] = []
quicksort (p:xs) = (quicksort lesser) ++ [p] ++ (quicksort greater)
    where
        lesser = filter (< p) xs
        greater = filter (>= p) xs

{-

In Detail

quicksort [] = []

means that the empty list yields the empty list.

quicksort (p:xs)

 means that the input must be a non-empty list where

  p is the head element and
  xs is the tail

 (p:xs) is a list pattern (a bit similar to regular expressions/patterns)


Let's consider the expression

 filter (< p) xs

Implies that the list elements must be comparable.
Hence, we required "Ord"




-}

-- The "cons" notation (meaning ":") can be used to observe
-- the input patterns.
-- (x:xs) is a list pattern where x refers to the head and xs to the tail.
-- List patterns match values.
-- For example, if we match (x:xs) against [1,2,3] we obtain
-- x is bound to 1 and xs is bound to [2,3]
--
-- There's no black magic.
-- [1,2,3] corresponds to 1 : (2 : [])
-- Haskell performs simply structural matching to compute the values of
-- pattern variables such as x and xs
myHead (x:xs) = x

myHead2 = \(x:xs) -> x

myHead3 = \y -> case y of
                  (x:xs) -> x

-- Convention, use don't care pattern variables
myHead4 (x:_) = x


-- myHead is  partial function, can only be applied to non-empty lists

-- extract the last element in a list
-- There are no for/while loops in Haskell, we loop via recursion.
myLast [x] = x
myLast (_:xs) = myLast xs


myLast2 = \y -> case y of
                  [x] -> x
                  (_:xs) -> myLast2 xs


-- The order of list patterns matters!

-- Patterns are tried in top to bottom order
-- In the below, we first check [x] and the (_:xs)
-- That means, if we reach (_:xs) the input list must consistent of
-- at least two elements.
myLast3 [x] = x
myLast3 (_:xs) = myLast3 xs

-- Some bug!
-- Case (2) will never be reached, because (_:xs) covers
-- singleton lists!
myLast4 (_:xs) = myLast4 xs    -- (1)
myLast4 [x] = x                -- (2)

-- Let's write a calculator for arithmetic expressions.

{-

Let's pick Java.

class Exp {
  int eval();
  Exp simp(); // adding functionality requires
              // to recompile all child classes
}

class Num extend Exp {
  int n;
  Num(int m) { n = m;}
  int eval() { return n; }
}

class Plus extend Exp {
  Exp l;
  Exp r;
  Plus(Exp x, Exp y) { l = x; r = y; }
  int eval() { return l.eval() + r.eval(); }
}

-- Every Mult object behaves like Exp
-- Mult <= Exp
class Mult extend Exp {
  Exp l;
  Exp r;
  Plus(Exp x, Exp y) { l = x; r = y; }
  int eval() { return l.eval() * r.eval(); }
}

// Adding cases is easy
class Div {
  Exp l;
  Exp r;
  Plus(Exp x, Exp y) { l = x; r = y; }
  int eval() { return l.eval() / r.eval(); }

}

-- Source syntax

1 + 3

  eval((1+3) + 4)

= eval(1+3) + eval(4)
= 4 + 4
= 8

-- Internal syntax

 "(1+3) + 4)"

represented by

   Plus(Plus(Num(1),Num(3)), Num(4)).eval

=  Plus(Num(1),Num(3)).eval + Num(4).eval
=  (Num(1).eval + Num(3).eval) + 4
=  (1+3)+4
=  4+4
=  8

-- Let's try the above in Haskell

data Exp = Val Int
         | Plus Exp Exp
         | Mult Exp Exp
-- If we add a new case,
-- we need to recompile/change all functions
-- that operate on Exp values.

instance Show Exp where
    show (Val i) = show i
    show (Plus e1 e2) = "(" ++ show e1 ++ " + " ++ show e2 ++ ")"
    show (Mult e1 e2) = "(" ++ show e1 ++ " * " ++ show e2 ++ ")"

eval :: Exp -> Int
eval (Val i) = i
eval (Mult e1 e2) = eval e1 + eval e2
eval (Plus e1 e2) = eval e1 * eval e2

-- Adding new functionality is easy.
-- Apply the law 0 * e = 0
simp :: Exp -> Exp
simp (Mult (Val 0) e) = Val 0
simp f = f

simp2 :: Exp -> Exp
simp2 e = case e of
            Mult e1 e2 -> case e1 of
                            Val 0 -> Val 0
                            _     -> e
            _ -> e

Evaluator, Type-Checker, and GADTs to enforce well-typing by construction

{-# LANGUAGE GADTs #-}


data E = One | Zero | ETrue | EFalse
       | Plus E E | Mult E E
       | Or E E




instance Show E where
    show One = "1"
    show Zero = "0"
    show ETrue = "true"
    show EFalse = "false"
    show (Plus e1 e2) = "(" ++ show e1 ++ " + " ++ show e2 ++ ")"
    show (Mult e1 e2) = "(" ++ show e1 ++ " * " ++ show e2 ++ ")"
    show (Or e1 e2) = "(" ++ show e1 ++ " || " ++ show e2 ++ ")"


eval :: E -> Maybe (Either Int Bool)
eval One = Just (Left 1)
eval Zero = Just (Left 0)
eval ETrue = Just (Right True)
eval EFalse = Just (Right False)
eval (Plus e1 e2) =
     let r1 = eval e1
         r2 = eval e2
     in case (r1, r2) of
         (Just (Left i1), Just (Left i2)) -> Just (Left (i1 + i2))
         (_,_)  -> Nothing
eval (Mult e1 e2) =
     let r1 = eval e1
         r2 = eval e2
     in case (r1, r2) of
         (Just (Left i1), Just (Left i2)) -> Just (Left (i1 * i2))
         (_,_)  -> Nothing
eval (Or e1 e2) =
     case (eval e1) of
        Nothing -> Nothing
        (Just Left{}) -> Nothing
        (Just (Right True)) -> Just (Right True)
        _                   -> case (eval e2) of
                                 Nothing -> Nothing
                                 (Just Left{}) -> Nothing
                                 (Just (Right True)) -> Just (Right True)
                                 _                   -> Just (Right False)

-- Running the evaluator and producing some output messages.
run e = case eval e of
          Nothing -> "Sorry failure"
          Just (Left i) -> "The Integer result is " ++ show i
          Just (Right b) -> "The Boolean result is " ++ show b


data Type = TInt | TBool deriving Show

typecheck :: E -> Maybe Type
typecheck Zero = Just TInt
typecheck One = Just TInt
typecheck ETrue = Just TBool
typecheck EFalse = Just TBool
typecheck (Plus e1 e2) =
     case (typecheck e1, typecheck e2) of
       (Just TInt, Just TInt) -> Just TInt
       (_, _) -> Nothing
typecheck (Mult e1 e2) =
     case (typecheck e1, typecheck e2) of
       (Just TInt, Just TInt) -> Just TInt
       (_, _) -> Nothing
typecheck (Or e1 e2) =
     case (typecheck e1, typecheck e2) of
       (Just TBool, Just TBool) -> Just TBool
       (_, _) -> Nothing


-- Assumption: E is type correct
evalT :: E -> Either Int Bool
evalT One = Left 1
evalT Zero = Left 0
evalT ETrue = Right True
evalT EFalse = Right False
evalT (Plus e1 e2) =
    case (evalT e1, evalT e2) of
       (Left i1, Left i2) -> Left (i1 + i2)
       -- What about the following case?
       -- (Right _, Right _) -> ???
       -- The above is dead code (under the assumption that we have applied typecheck).
       -- Therefore, "Left" feels redundant!
       -- But it is necessary, because values of type E might be ill-typed.
       -- At the "Haskell level", there is no information that E is type correct.
evalT (Mult e1 e2) =
    case (evalT e1, evalT e2) of
       (Left i1, Left i2) -> Left (i1 * i2)
evalT (Or e1 e2) =
     case (evalT e1) of
       Right True -> Right True
       _          -> evalT e2


runTypeCheckAndEvaluate e =
     case typecheck e of
       Nothing -> "type failure"
       _       -> case evalT e of
                    Left i -> "the Integer result is " ++ show i
                    Right b -> "the Boolean result is " ++ show b


-- Rephrase E in terms GADT notation. I call the resulting data type EE
data EE where
    One_EE :: EE
    Zero_EE :: EE
    ETrue_EE :: EE
    EFalse_EE :: EE
    Plus_EE :: EE -> EE -> EE
    Mult_EE :: EE -> EE -> EE
    Or_EE :: EE -> EE -> EE
    deriving Show

-- I'm an expression and my type is 'a'.
data Exp a where
   One_Exp :: Exp Int
   Zero_Exp :: Exp Int
   ETrue_Exp :: Exp Bool
   EFalse_Exp :: Exp Bool
   Plus_Exp :: Exp Int -> Exp Int -> Exp Int
   Mult_Exp :: Exp Int -> Exp Int -> Exp Int
   Or_Exp :: Exp Bool -> Exp Bool -> Exp Bool

-- Any value constructed by "Exp a" is well-typed!

-- Difference to evalT is as follows.
-- By using "Exp a" we know the expression is well-typed.
evalExp :: Exp a -> a
evalExp One_Exp = 1
evalExp Zero_Exp = 1
evalExp ETrue_Exp = True
evalExp EFalse_Exp = False
evalExp (Plus_Exp e1 e2) = evalExp e1 + evalExp e2
evalExp (Mult_Exp e1 e2) = evalExp e1 * evalExp e2
evalExp (Or_Exp e1 e2) = evalExp e1 || evalExp e2

data List a = Nil | Cons a (List a)

member :: Eq a => a -> List a -> Bool
member x Nil = False
member x (Cons y ys)
    | x == y    = True
    | otherwise = member x ys

data DictEq a = DictEqVal (a->a->Bool)
unDict (DictEqVal f) = f

-- During translation, predicate "Eq a" is is turned into the dictionary "DictEq a".
memberDict :: DictEq a -> a -> List a -> Bool
memberDict d x Nil = False
memberDict (DictEqVal eq) x (Cons y ys)
    | eq x y    = True
    | otherwise = memberDict (DictEqVal eq) x ys

-- Haskell is expression-oriented and pure.
-- Whatever happens, is reflected in the type!

example1 :: Int -> Int
example1 x = let y = x
             in x + y

apply :: (Int -> Int) -> Int     -- more specific but valid type
apply f = (f 3) + 4      -- f is assumed to be a (pure) function

example2 :: Int -> IO Int            -- Why IO?
                                     -- The program text is no longer a pure expression.
                                     -- We execute a sequence of commands!
                                     -- Haskell is very precise about the "things" (side effects) that are happening.
                                     -- IO Int means the following
                                     --   We obtain some Int value.
                                     --   The "computation" of the Int value involves some side effect.
                                     --   And this is reflected in the type.
example2 x = do { let y = x
                ;    print x
                ; z <- getLine       -- why not z = getLine?
                                     -- In detail, the call "getLine" involves some "side effects",
                                     -- needs to interact with the "system".
                                     -- This means that "getLine" is not pure.
                                     -- For side effecting computations that yield a value and
                                     -- that value shall be assigned to a variable,
                                     -- Haskell uses the "<-" symbol.
                                     -- Details follow!
                ; print z
            ; return (x + y)
                }

-- Can get rid of "{}" and ";".
-- Be careful, layout matters (due to some desugaring step).
example2b x = do let y = x
                 print x
                 z <- getLine
                 print z
                 return (x + y)

Consider the following C/Java statement

1. First observation, the left and the right x are different!

2. Here's how to interpret the above statement.

   We compute "x + 1" based on the current state S1 (mapping of variables to values).

   The we update the state S1 and obtain a new state S2 where
      S2(x) = S1(x) + 1  and S2(y) = S1(y) for all y != x

 None of this is visible in the source program, the C compiler + run-time do the "hard" work for us.

\s -> let value = lookup(s,x) + 1

                                   -- all other y's are unchanged

  Assignment in C/Java is a state transformer!

   We need to "embed" imperative Code (state transformers) into Haskell!

-- Library for test data generators

class Arbitrary a where
   arbitrary :: IO a

-- | Choose one of the elements.
elements :: [a] -> IO a
elements xs = do i <- randomIO :: IO Int
                 return (xs !! (i `mod` (length xs)))

-- | Generate a fixed number of arbitrary values.
vector ::  Arbitrary a => Int -> IO [a]
vector 0 = return []
vector n
    | n > 0  = do x <- arbitrary
                  xs <- vector (n-1)
                  return (x:xs)
    | otherwise = error "impossible"


instance Arbitrary Char where
   arbitrary = do x <- elements [0..255]
                  return (chr x)

instance Arbitrary a => Arbitrary [a] where
   arbitrary = vector 5

instance (Arbitrary a, Arbitrary b) => Arbitrary (Either a b) where
   arbitrary = do b <- randomIO
                  if b
                    then do l <- arbitrary
                            return $ Left l
                    else do r <- arbitrary
                            return $ Right r


-- | Random generation of strings
genStrings :: IO ()
genStrings = do xs <- arbitrary -- xs must be a String ([Char])
                putStrLn xs     -- because applied here

-- | Quickly check a property by testing the property
-- against 100 arbitrarily generated test inputs.
quickCheck :: (Show t, Arbitrary t) => (t -> Bool) -> IO ()
quickCheck prop = go 300
   where go 0 = putStrLn "+++ Ok"
         go n = do x <- arbitrary
                   if prop x
                     then go (n-1)
                     else do putStrLn "*** Failed: "
                             print x

import Test.QuickCheck

-- QuickCheck for test data generation

-- Binary trees
data Tree a = Leaf a | Node (Tree a) (Tree a) deriving (Show, Eq)

-- Arbitrary instance
instance Arbitrary a => Arbitrary (Tree a) where
    arbitrary = do left <- arbitrary
                   right <- arbitrary
                   x <- arbitrary
                   r <- elements [Leaf x, Node left right]
                   return r
